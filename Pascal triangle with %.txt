public class PascalTriangle {
    public static void main(String[] args) {
        int rows = 3; 

        for (int i = 0; i < rows; i++) {
           
            for (int j = rows - i - 1; j > 0; j--) {
                System.out.print(" ");
            }
            for (int j = 0; j <= i; j++) {
                System.out.print(" % ");
            }
            
            System.out.println(); 
        }
    }
}
